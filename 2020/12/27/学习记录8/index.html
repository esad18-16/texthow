传智杯初赛题1：
题目：
解题思路：设置两个参数j、num，j用来记录已经报名了多少个学员到达m个时回到0，num记录费用，直接使用for循环以n为结束标志，当j与m相等时价格v+=a。
AC代码如下：


#include
int main(void)
{
    int n,v,m,a;
    while(scanf("%d %d %d %d",&n,&v,&m,&a)!=EOF)
    {
        int i,j=0,num=0;
        for(i=0;i        {
            if(j==m)
            {
                v+=a;
                j=0;
            }
            num+=v;
            j++;
        }
        printf("%d\n",num);
     }
    return 0;
 }

传智杯初赛题2
题目：
解题思路：
直接使用if语句判断即可，分为三种情况，首先是成绩大于等于90的输出4.0，然后时60到89的，输出4.0-(90-成绩)*0.1即可，最后是小于60分的，由于根据题目要求开根后乘以10，这种情况不存在大于等90分的，直接判断开根后乘以10是否大于等于60即可。

AC代码如下：


#include
#include
int main(void)
{
    int n;
    while(scanf("%d",&n)!=EOF)
    {
        int num=0;
        if(n>=90)
        {
            printf("4.0\n");
            continue;
        }
        if(n>=60&&n<=89)
        {
            printf("%.1f\n",4.0-(90-n)*0.1);
        }
        if(n<60)
        {
            num=(int)(sqrt(n)*10);                    //强制转化为int型直接去掉小数点后的数
            if(num>=60)                                  //实现题目要求的向下取整
            printf("%.1f\n",4.0-(90-num)*0.1);
            else printf("0.0\n");
        }
    }
    return 0;
 }
